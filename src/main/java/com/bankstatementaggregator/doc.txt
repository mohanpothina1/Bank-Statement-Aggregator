while (running && socket != null && !socket.isClosed()) {

    long now = System.currentTimeMillis();

    //  If no data for 3 seconds → treat as LAN disconnected
    if (now - lastDataTime > 3000) {

        System.out.println("No data from " + ip + " → forcing reconnect");

        connected.set(false);
        updateLanStatus(false); //  make RED

        closeSocket(); // CLOSE SOCKET
        break; //  EXIT LOOP → auto reconnect happens
    }

    Thread.sleep(200);
}

connected.set(true);
lastDataTime = System.currentTimeMillis(); // ✅ ADD THIS
System.out.println("Connected to " + ip + ":" + port);

private void handleFrame(byte[] frame) {

    if (!verifyXor(frame)) {
        return;
    }

    HardwareData data = parseFrame(frame);

    this.lastData = data;
    this.lastDataTime = System.currentTimeMillis(); // ✅ IMPORTANT

    if (dataCallback != null) {
        dataCallback.accept(data);
    }

    System.out.println("VSWR: " + data.vswr +
            " TEMP: " + data.temp +
            " FWD: " + data.fwd +
            " RFL: " + data.rfl);
}private volatile long lastDataTime = 0;
